68 """Create preprocessing pipeline based on config.""" 

69 msg = Printer(timestamp=True) 

70 

71 steps = [] 

72 # Imputation 

73 if ( 

74 cfg.model.require_imputation 

75 and not cfg.preprocessing.post_split.imputation_method 

76 ): 

77 msg.warn( 

78 f"{cfg.model.name} requires imputation, but no imputation method was specified in the config file. Overriding to 'mean'.", 

79 ) 

80 

81 steps.append( 

82 ( 

83 "Imputation", 

84 SimpleImputer(strategy="mean"), 

85 ), 

86 ) 

87 # Not a great solution, but preferable to the script breaking and stopping a hyperparameter search. 

88 

89 if cfg.preprocessing.post_split.imputation_method: 

90 steps.append( 

91 ( 

92 "Imputation", 

93 SimpleImputer(strategy=cfg.preprocessing.post_split.imputation_method), 

94 ), 

95 ) 

96 

97 # Feature selection 

98 # Important to do this before scaling, since chi2 

99 # requires non-negative values

102 # Feature scaling 

103 # Important to do this after feature selection, since

106 if cfg.preprocessing.post_split.scaling: 

107 if cfg.preprocessing.post_split.scaling in { 

108 "z-score-normalization", 

109 "z-score-normalisation",

111 steps.append( 

112 ("z-score-normalization", StandardScaler()),

115 raise ValueError( 

116 f"{cfg.preprocessing.post_split.scaling} is not implemented. See above", 

117 ) 

118 

119 return Pipeline(steps)
