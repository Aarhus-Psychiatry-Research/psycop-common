17def text_preprocessing( 

18 df: pd.DataFrame, 

19 text_column_name: str = "value", 

20) -> pd.DataFrame: 

21 """Preprocess texts by lower casing, removing stopwords and symbols. 

22 

23 Args: 

24 df (pd.DataFrame): Dataframe with a column containing text to clean. 

25 text_column_name (str): Name of column containing text. Defaults to "value". 

26 

27 Returns: 

28 pd.DataFrame: df with preprocessed text 

29 """ 

30 # Define regex for stop words with empty string in beginning and end 

31 regex_stop_words_surrounded_by_empty_strings = [ 

32 rf"\b{stop_word}\b" for stop_word in stop_words 

33 ] 

34 regex_stop_words = "|".join(regex_stop_words_surrounded_by_empty_strings) 

35 

36 # Define regex that removes symbols (by keeping everything else) 

37 regex_symbol_removal = r"[^ÆØÅæøåA-Za-z0-9 ]+" 

38 

39 # combine 

40 regex_symbol_removal_and_stop_words = re.compile( 

41 f"{regex_stop_words}|{regex_symbol_removal}", 

42 ) 

43 

44 # lower case and remove stop words and symbols 

45 df[text_column_name] = ( 

46 df[text_column_name] 

47 .str.lower() 

48 .replace(regex_symbol_removal_and_stop_words, value="", regex=True) # type: ignore 

49 ) 

50 

51 return df
