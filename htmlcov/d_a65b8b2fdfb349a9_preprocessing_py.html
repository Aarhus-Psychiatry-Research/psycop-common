16def text_preprocessing( 

17 df: pd.DataFrame, 

18 text_column_name: str = "value", 

19) -> pd.DataFrame: 

20 """Preprocess texts by lower casing, removing stopwords and symbols. 

21 

22 Args: 

23 df (pd.DataFrame): Dataframe with a column containing text to clean. 

24 text_column_name (str): Name of column containing text. Defaults to "value". 

25 

26 Returns: 

27 pd.DataFrame: df with preprocessed text 

28 """ 

29 # Define regex for stop words with empty string in beginning and end 

30 regex_stop_words_surrounded_by_empty_strings = [ 

31 rf"\b{stop_word}\b" for stop_word in stop_words 

32 ] 

33 regex_stop_words = "|".join(regex_stop_words_surrounded_by_empty_strings) 

34 

35 # Define regex that removes symbols (by keeping everything else) 

36 regex_symbol_removal = r"[^ÆØÅæøåA-Za-z0-9 ]+" 

37 

38 # combine 

39 regex_symbol_removal_and_stop_words = re.compile( 

40 f"{regex_stop_words}|{regex_symbol_removal}", 

41 ) 

42 

43 # lower case and remove stop words and symbols 

44 df[text_column_name] = ( 

45 df[text_column_name] 

46 .str.lower() 

47 .replace(regex_symbol_removal_and_stop_words, value="", regex=True) # type: ignore 

48 ) 

49 

50 return df
